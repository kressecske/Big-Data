import matplotlib.pyplot as plt
%matplotlib inline

def list_of_regions(df):
    return df.region.unique()
def life_in_region(df, region):
    tmp = df[df.region == region]
    return tmp[['lifeMale', 'lifeFemale']].mean(axis=1)
    
for region in list_of_regions(df):
    plt.figure(figsize=(16,2))
    plt.xlim((0,100))
    plt.xlabel('Age')
    plt.ylabel('P')
    life_in_region(df, region).hist(bins=range(0,100,5), label=region, alpha=0.5, normed=True)
    plt.legend()
    plt.show()

from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import silhouette_score

def generate_models(K_array, train):
    models = []
    for i in K_array:
        kmeans = KMeans(n_clusters=i, max_iter=100)
        kmeans.fit(train)
        models.append(kmeans)
    return models

# NOTE: Use 100 sample points to calculate the silhouette score
def get_silhouette_scores(train, model_array):
    scores = []
    for m in model_array:
        scores.append(silhouette_score(train, m.predict(train), sample_size=100))
    return scores

K_array = range(2,11,1)
